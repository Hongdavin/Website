Excelsior Mobile SQL Report

Business intelligence report for a mobile carrier using T-SQL. Queried subscriber usage, billing, device, and plan data across multiple joined tables to surface customer insights, marketing opportunities, and usage patterns by city and plan.

-- Customer usage + billing summary
SELECT CONCAT(S.FirstName + SPACE(1), S.LastName) AS 'Full Name',
       Minutes AS 'Minute Usage',
       DataInMB AS 'Data Usage',
       Texts AS 'Text Usage',
       B.Total AS 'Total Bill'
FROM Subscriber AS S, Bill AS B, LastMonthUsage AS L
WHERE S.MIN = B.MIN
  AND S.MIN = L.MIN
ORDER BY 'Full Name'

-- AVG usage + billing by city

       AVG(Total) AS 'AVG Total Bill'
FROM Bill AS B, LastMonthUsage AS L, Subscriber AS S
WHERE S.MIN = L.MIN AND S.MIN = B.MIN
GROUP BY City

def maze_solver(maze):
    """
    Simulate a light ray traveling through a mirror maze.
    maze: 2D list where 1 = '/' mirror, -1 = '\\' mirror, 0 = empty
    """
    rows = len(maze)
    cols = len(maze[0])
    row, col = 0, 0
    dr, dc = 0, 1
    path = []
    max_steps = rows * cols * 2

    for _ in range(max_steps):
        if row < 0 or row >= rows or col < 0 or col >= cols:
            break
        path.append((row, col))
        cell = maze[row][col]

        if cell == 1:
            dr, dc = -dc, -dr
        elif cell == -1:
            dr, dc = dc, dr

        row += dr
        col += dc

    return {"path": path, "exit": (row, col), "steps": len(path)}

def findMinRooms(*meetings):
    """
    Calculate minimum conference rooms needed.
    Each meeting is a tuple (start, end) in 24-hour float format.
    """
    if not meetings:
        return 0

    starts = []
    ends   = []

    for meeting in meetings:
        if not isinstance(meeting, (tuple, list)) or len(meeting) != 2:
            raise TypeError(f"Invalid meeting format: {meeting}")
        start, end = meeting
        if not (0.0 <= start <= 24.0) or not (0.0 <= end <= 24.0):
            raise ValueError(f"Time out of range (0-24): {meeting}")
        if start >= end:
            raise ValueError(f"Start must be before end: {meeting}")
        starts.append(start)
        ends.append(end)

    starts.sort()
    ends.sort()

    current_rooms = max_rooms = s_ptr = e_ptr = 0

    while s_ptr < len(starts):
        if starts[s_ptr] < ends[e_ptr]:
            current_rooms += 1
            max_rooms = max(max_rooms, current_rooms)
            s_ptr += 1
        else:
            current_rooms -= 1
            e_ptr += 1

    return max_rooms

import pandas as pd
import os

stocks = ["IBM", "MSFT", "GOOG", "AAPL", "AMZN", "FB"]

def analyze_dividends(symbol):
    """
    Detect dividend payments from historical stock price data.
    Dividends inferred when Adj Close diverges from Close.
    """
    df = pd.read_csv(f"{symbol}.csv")
    df.columns = [c.strip() for c in df.columns]
    df["Date"]  = pd.to_datetime(df["Date"])
    df = df.sort_values("Date").reset_index(drop=True)

    # Day-over-day ratios
    df["close_ratio"] = df["Close"].shift(1)     / df["Close"]
    df["adj_ratio"]   = df["Adj Close"].shift(1) / df["Adj Close"]

    # Dividend = ratio divergence × closing price
    df["Dividend"] = (
        (df["adj_ratio"] - df["close_ratio"]) * df["Close"]
    ).round(4)

    dividends = df[df["Dividend"].abs() > 0.01][
        ["Date", "Close", "Adj Close", "Dividend"]
    ]

    return {
        "symbol":    symbol,
        "dividends": dividends,
        "total":     round(dividends["Dividend"].sum(), 4),
        "count":     len(dividends),
    }

for symbol in stocks:
    result = analyze_dividends(symbol)
    if result:
        print(f"{result['symbol']}: {result['count']} payments  ${result['total']}")
