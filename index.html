KNN House Price Prediction Model

R Machine Learning Project

# Load necessary libraries
library(caret); library(ggplot2); library(ROSE); library(readr); library(dplyr)

# Load and clean data
house <- read.csv("house_14.csv") %>%
  mutate(price = ifelse(price <= mean(price, na.rm = TRUE), "Low", "High")) %>%
  na.omit() %>%
  select(bedrooms, bathrooms, sqft_living, sqft_lot, floors, 
         waterfront, view, condition, grade, sqft_above, price) %>%
  mutate(price = as.factor(price), waterfront = as.factor(waterfront))

# Train-test split (60-40)
set.seed(666)
train_index <- createDataPartition(house$price, p = 0.6, list = FALSE)
train <- house[train_index, ]; valid <- house[-train_index, ]

# Balance dataset using ROSE
train <- ROSE(price ~ ., data = train, seed = 666)$data
valid <- ROSE(price ~ ., data = valid, seed = 666)$data

# Normalize data
norm_values <- preProcess(train[, -ncol(train)], method = c("center", "scale"))
train_norm <- predict(norm_values, train)
valid_norm <- predict(norm_values, valid)

# Train KNN models with k=5 and k=7
knn_k5 <- knn3(price ~ ., data = train_norm, k = 5)
knn_k7 <- knn3(price ~ ., data = train_norm, k = 7)

# Evaluate models
pred_k5 <- predict(knn_k5, valid_norm[, -ncol(valid_norm)], type = "class")
pred_k7 <- predict(knn_k7, valid_norm[, -ncol(valid_norm)], type = "class")

cm_k5 <- confusionMatrix(pred_k5, valid_norm$price, positive = "High")
cm_k7 <- confusionMatrix(pred_k7, valid_norm$price, positive = "High")

cat("K=5 Validation Accuracy:", cm_k5$overall["Accuracy"], "\n")
cat("K=7 Validation Accuracy:", cm_k7$overall["Accuracy"], "\n")

# Select best model
best_model <- if(cm_k7$overall["Accuracy"] > cm_k5$overall["Accuracy"]) knn_k7 else knn_k5

# Predict on new house data
new_house <- read_csv("house_test_14.csv") %>%
  select(bedrooms, bathrooms, sqft_living, sqft_lot, floors, 
         waterfront, view, condition, grade, sqft_above) %>%
  mutate(waterfront = as.factor(waterfront))

new_house_norm <- predict(norm_values, new_house)
predictions <- predict(best_model, new_house_norm, type = "class")
print(predictions)

def maze_solver(maze):
    """Solve the mirror maze and return the path of the light ray."""
    row, col = 0, 0
    dr, dc = 0, 1  # Start moving right
    path = []

    while True:
        path.append((row, col))
        cell = maze[row][col]

        # Update direction if hitting a mirror
        if cell == 1:  # 45-degree mirror ('/')
            if dr == -1 and dc == 0:   dr, dc = 0, 1   # up -> right
            elif dr == 1 and dc == 0:  dr, dc = 0, -1  # down -> left
            elif dr == 0 and dc == -1: dr, dc = 1, 0   # left -> down
            elif dr == 0 and dc == 1:  dr, dc = -1, 0  # right -> up

        elif cell == -1:  # -45-degree mirror ('\')
            if dr == -1 and dc == 0:   dr, dc = 0, -1  # up -> left
            elif dr == 1 and dc == 0:  dr, dc = 0, 1   # down -> right
            elif dr == 0 and dc == -1: dr, dc = -1, 0  # left -> up
            elif dr == 0 and dc == 1:  dr, dc = 1, 0   # right -> down

        # Move to next cell
        row += dr
        col += dc

        # Stop if light exits the maze
        if row < 0 or col < 0 or row >= len(maze) or col >= len(maze[0]):
            break

    return path

Conference Room Scheduler Optimizer

Python Scheduling Algorithm - Interactive Demo

Try It Yourself!

Source Code:

def findMinRooms(meetings):
    """Calculate minimum conference rooms needed."""
    if not meetings:
        return "Error: No meetings provided"
    
    starts, ends = [], []
    
    for idx, meeting in enumerate(meetings, start=1):
        try:
            start, end = meeting
            
            if not (isinstance(start, (int, float)) and isinstance(end, (int, float))):
                return f"Error: Meeting #{idx} contains non-numeric values"
            
            if not (0.0 <= start <= 24.0 and 0.0 <= end <= 24.0):
                return f"Error: Meeting #{idx} times must be 0.0-24.0"
            
            if start >= end:
                return f"Error: Meeting #{idx} start must be < end"
            
            starts.append(start)
            ends.append(end)
        except:
            return f"Error: Meeting #{idx} format invalid"
    
    starts.sort()
    ends.sort()
    
    current_rooms = max_rooms = 0
    i = j = 0
    
    while i < len(meetings):
        if starts[i] < ends[j]:
            current_rooms += 1
            max_rooms = max(max_rooms, current_rooms)
            i += 1
        else:
            current_rooms -= 1
            j += 1
    
    return max_rooms

# Example usage:
print(findMinRooms([[1.2, 3.4], [2.3, 5.0], [3.1, 8.0]]))  # Output: 3

# Calculate implied dividends from price adjustments
def calculate_dividends(prices):
    dividends = []
    
    for i in range(1, len(prices)):
        prev_close = prices[i-1]['close']
        curr_close = prices[i]['close']
        prev_adj = prices[i-1]['adj_close']
        curr_adj = prices[i]['adj_close']
        
        # Calculate ratios
        close_ratio = prev_close / curr_close
        adj_ratio = prev_adj / curr_adj
        
        # Dividend formula
        dividend = (adj_ratio - close_ratio) * curr_close
        
        if abs(dividend) > 0.01:  # Filter out negligible amounts
            dividends.append({
                'date': prices[i]['date'],
                'amount': round(dividend, 2)
            })
    
    return dividends

# Example: Apple stock with dividend payment
prices = [
    {'date': '2024-02-09', 'close': 187.68, 'adj_close': 187.23},
    {'date': '2024-02-12', 'close': 188.85, 'adj_close': 188.39}
]

result = calculate_dividends(prices)
# Output: Dividend of ~$0.24 per share detected on 2024-02-12
