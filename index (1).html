def maze_solver(maze):
    """
    Simulate a light ray traveling through a mirror maze.
    maze: 2D list where 1 = '/' mirror, -1 = '\\' mirror, 0 = empty
    Light enters from top-left moving right.
    Returns the exit coordinates and full path.
    """
    rows = len(maze)
    cols = len(maze[0])
    row, col = 0, 0
    dr, dc = 0, 1   # direction: right
    path = []
    max_steps = rows * cols * 2  # prevent infinite loops

    for _ in range(max_steps):
        if row < 0 or row >= rows or col < 0 or col >= cols:
            break  # exited the maze
        path.append((row, col))
        cell = maze[row][col]

        if cell == 1:    # '/' mirror
            dr, dc = -dc, -dr
        elif cell == -1: # '\\' mirror
            dr, dc = dc, dr

        row += dr
        col += dc

    return {
        "path": path,
        "exit": (row, col),
        "steps": len(path)
    }

def findMinRooms(*meetings):
    """
    Calculate minimum conference rooms needed for a set of meetings.
    Each meeting is a tuple (start, end) in 24-hour float format.
    Raises ValueError for invalid times or malformed input.
    """
    if not meetings:
        return 0

    starts = []
    ends   = []

    for meeting in meetings:
        if not isinstance(meeting, (tuple, list)) or len(meeting) != 2:
            raise TypeError(f"Invalid meeting format: {meeting}")
        start, end = meeting
        if not (0.0 <= start <= 24.0) or not (0.0 <= end <= 24.0):
            raise ValueError(f"Time out of range (0-24): {meeting}")
        if start >= end:
            raise ValueError(f"Start must be before end: {meeting}")
        starts.append(start)
        ends.append(end)

    starts.sort()
    ends.sort()

    current_rooms = 0
    max_rooms     = 0
    s_ptr = 0
    e_ptr = 0

    while s_ptr < len(starts):
        if starts[s_ptr] < ends[e_ptr]:
            current_rooms += 1
            max_rooms = max(max_rooms, current_rooms)
            s_ptr += 1
        else:
            current_rooms -= 1
            e_ptr += 1

    return max_rooms

import pandas as pd
import os

stocks = ["IBM", "MSFT", "GOOG", "AAPL", "AMZN", "FB"]

def analyze_dividends(symbol):
    """
    Detect dividend payments from historical stock price data.
    Dividends are inferred when Adj Close diverges from Close.
    """
    filepath = f"{symbol}.csv"
    if not os.path.exists(filepath):
        print(f"File not found: {filepath}")
        return None

    df = pd.read_csv(filepath)
    df.columns = [c.strip() for c in df.columns]
    df["Date"] = pd.to_datetime(df["Date"])
    df = df.sort_values("Date").reset_index(drop=True)

    # Compute day-over-day ratios
    df["close_ratio"] = df["Close"].shift(1) / df["Close"]
    df["adj_ratio"]   = df["Adj Close"].shift(1) / df["Adj Close"]

    # Dividend = difference in ratios * closing price
    df["Dividend"] = (df["adj_ratio"] - df["close_ratio"]) * df["Close"]
    df["Dividend"]  = df["Dividend"].round(4)

    # Filter rows where a dividend was detected
    dividends = df[df["Dividend"].abs() > 0.01][["Date", "Close", "Adj Close", "Dividend"]]

    total = dividends["Dividend"].sum()
    return {
        "symbol":    symbol,
        "dividends": dividends,
        "total":     round(total, 4),
        "count":     len(dividends)
    }

# Run analysis for all stocks
for symbol in stocks:
    result = analyze_dividends(symbol)
    if result:
        print(f"\n{'='*40}")
        print(f"Symbol: {result['symbol']}")
        print(f"Dividend payments detected: {result['count']}")
        print(f"Total dividends per share:  ${result['total']}")
        print(result["dividends"].to_string(index=False))
